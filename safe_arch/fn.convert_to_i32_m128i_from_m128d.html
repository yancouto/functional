pub fn convert_to_i32_m128i_from_m128d(a: m128d) -> m128i

Rounds the two f64 lanes to the low two i32 lanes.

let a = m128d::from_array([1.0, 2.5]);
let b = convert_to_i32_m128i_from_m128d(a);
let c: [i32; 4] = b.into();
assert_eq!(c, [1, 2, 0, 0]);
